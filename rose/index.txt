    function loadGlobalReviews() {
        const saved = localStorage.getItem("rose_team_global_reviews");
        if (saved) {
            globalReviews = JSON.parse(saved);
        } else {
            globalReviews = generateGlobalReviews();
            saveGlobalReviews();
        }
    }
    
    function saveGlobalReviews() {
        localStorage.setItem("rose_team_global_reviews", JSON.stringify(globalReviews));
    }
    
    // Загрузка локальных отзывов пользователя
    function loadUserReviews() {
        const saved = localStorage.getItem("rose_team_user_reviews_" + (window.location.hostname + window.location.pathname));
        if (saved) {
            userReviews = JSON.parse(saved);
        } else {

        }
    }
    
    function saveUserReviews() {
        localStorage.setItem("rose_team_user_reviews_" + (window.location.hostname + window.location.pathname), JSON.stringify(userReviews));
    }
    
    // Добавление отзыва пользователем
    function addUserReview(name, text) {
        const newReview = {
            id: Date.now(),
            name: name.trim(),
            text: text.trim(),
            date: new Date().toLocaleDateString(),
            isAdmin: false,
            isGlobal: false,
            isUser: true
        };
        userReviews.unshift(newReview);

    }
